CREATE TABLE public.albums (
	id serial4 NOT NULL,
	namealbums varchar NOT NULL,
	releaseyear int4 NOT NULL,
	idartists int4 NOT NULL,
	CONSTRAINT albums_pk PRIMARY KEY (id),
	CONSTRAINT albums_id_fkey FOREIGN KEY (id) REFERENCES public.artist(id)
);

CREATE TABLE public.artist (
	id serial4 NOT NULL,
	nameartists varchar NULL,
	artists_genres varchar NULL,
	CONSTRAINT artist_pk PRIMARY KEY (id)
);

CREATE TABLE public.genres (
	id serial4 NOT NULL,
	namegenres varchar NULL,
	CONSTRAINT genres_pk PRIMARY KEY (id),
	CONSTRAINT genres_id_fkey FOREIGN KEY (id) REFERENCES public.artist(id)
);

CREATE TABLE public.tracks (
	id serial4 NOT NULL,
	nametracks varchar NOT NULL,
	duration time NOT NULL,
	idalbums int4 NOT NULL,
	CONSTRAINT tracks_pk PRIMARY KEY (id),
	CONSTRAINT tracks_id_fkey FOREIGN KEY (id) REFERENCES public.albums(id)
);
